JSON Format for HTTP Requests

POST (Create)

{
  "address": "str",
  "dob": "str",
  "email": "str",
  "password": "str",
  "image": "str",
  "name": "str",
  "phone": "str"
}

GET (Read)
{
  "address": "str",
  "dob": "str",
  "email": "str",
  "password": "str",
  "uuid": "str",
  "image": "str",
  "name": "str",
  "phone": "str"
}

PATCH (Update)

{
  "uuid": "str",
  "<update_feild>": "str"
}

DELETE (Delete)

{
  "uuid": "str"
}
